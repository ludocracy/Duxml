def initialize(*args)

  if class_to_xml *args

    args.first[:object].each do |k, v| @xml[k] = v end if args.first[:object].is_a?(Hash)

    @xml.remove_attribute 'object'

    @xml.remove_attribute 'old_value'

  end

  super()

end
